  7class mesh_vector_calculus:
  8    """Vector calculus on uw row matrices
  9    - this class is designed to augment the functionality of a mesh"""
 10
 11    def __init__(self, mesh):
 12        self.mesh = mesh
 13        self.dim = self.mesh.dim
 14
 15    def cross(self, vector1, vector2):
 16        a = self.to_vector(vector1)
 17        b = self.to_vector(vector2)
 18
 19        a_cross_b_vec = sympy.vector.cross(a, b)
 20        a_cross_b = self.to_matrix(a_cross_b_vec)
 21
 22        return a_cross_b
 23
 24    def curl(self, matrix):
 25        r"""
 26        $ \nabla \times \mathbf{v} $
 27
 28        Returns the curl of a 3D vector field or the out-of-plane
 29        component of a 2D vector field
 30        """
 31
 32        vector = self.to_vector(matrix)
 33        vector_curl = sympy.vector.curl(vector)
 34
 35        if self.dim == 3:
 36            return self.to_matrix(vector_curl)
 37        else:
 38            # if 2d, the out-of-plane vector is not defined in the basis so a scalar is returned (cf. vorticity)
 39            return vector_curl.dot(self.mesh.N.k)
 40
 41    def divergence(self, matrix):
 42        r"""
 43        $ \nabla \cdot \mathbf{v} $
 44        """
 45        vector = self.to_vector(matrix)
 46        scalar_div = sympy.vector.divergence(vector)
 47        return scalar_div
 48
 49    def gradient(self, scalar):
 50        r"""
 51        $\nabla \phi$
 52        """
 53
 54        if isinstance(scalar, sympy.Matrix) and scalar.shape == (1, 1):
 55            scalar = scalar[0, 0]
 56
 57        vector_gradient = sympy.vector.gradient(scalar)
 58        return self.to_matrix(vector_gradient)
 59
 60    def strain_tensor(self, vector):
 61        """
 62        Components of the infinitessimal strain or strain-rate tensor where the
 63        vector that is provided is displacement or velocity respectively
 64        """
 65
 66        # Coerce vector to sympy.Matrix form
 67        matrix = self.to_matrix(vector)
 68
 69        L = matrix.jacobian(self.mesh.CoordinateSystem.N)
 70        E = (L + L.transpose()) / 2
 71
 72        return E
 73
 74    def to_vector(self, matrix):
 75        if isinstance(matrix, sympy.vector.Vector):
 76            return matrix  # No need to convert
 77
 78        # Note, the mesh vector basis is always 3D so out-of-plane
 79        # vectors are allowed.
 80
 81        if matrix.shape == (1, 2) or matrix.shape == (1, 3):
 82            vector = sympy.vector.matrix_to_vector(matrix, self.mesh.N)
 83        elif matrix.shape == (2, 1) or matrix.shape == (3, 1):
 84            vector = sympy.vector.matrix_to_vector(matrix.T, self.mesh.N)
 85        elif matrix.shape == (1, 1):
 86            vector = matrix[0, 0]
 87        else:
 88            print(f"Unable to convert matrix of size {matrix.shape} to sympy.vector")
 89            vector = None
 90
 91        return vector
 92
 93    def to_matrix(self, vector):
 94        # Almost there ...
 95
 96        if isinstance(vector, sympy.Matrix) and vector.shape == (1, self.dim):
 97            return vector
 98
 99        if isinstance(vector, sympy.Matrix) and vector.shape == (self.dim, 1):
100            return vector.T
101
102        if isinstance(vector, sympy.Matrix) and vector.shape == (1, 1):
103            return vector
104
105        matrix = sympy.Matrix.zeros(1, self.dim)
106        base_vectors = self.mesh.N.base_vectors()
107
108        for i in range(self.dim):
109            matrix[0, i] = vector.dot(base_vectors[i])
110
111        return matrix
112
113    def jacobian(self, matrix):
114        """Jacobian of a quantity (scalar, vector, matrix) wrt the natural mesh coordinates"""
115
116        if isinstance(matrix, sympy.vector.Vector):
117            matrix_form = self.to_matrix(matrix)
118        else:
119            matrix_form = matrix
120
121        jac = matrix_form.jacobian(self.mesh.CoordinateSystem.N)
122
123        return jac

126class mesh_vector_calculus_cylindrical(mesh_vector_calculus):
127    """
128    mesh_vector_calculus module for div, grad, curl etc that apply in
129    native cylindrical coordinates
130    """
131
132    def __init__(self, mesh):
133        coordinate_type = mesh.CoordinateSystem.coordinate_type
134
135        # validation
136
137        if not (
138            coordinate_type == CoordinateSystemType.CYLINDRICAL2D_NATIVE
139            or coordinate_type == CoordinateSystemType.CYLINDRICAL3D_NATIVE
140        ):
141            print(
142                f"Warning mesh type {mesh.CoordinateSystem.type} uses Cartesian coordinates not cylindrical"
143            )
144
145        super().__init__(mesh)
146
147    def divergence(self, matrix):
148        r"""
149        $ \nabla \cdot \mathbf{v} $
150        """
151
152        r = self.mesh.CoordinateSystem.N[0]
153        t = self.mesh.CoordinateSystem.N[1]
154
155        V_r = matrix[0]
156        V_t = matrix[1]
157
158        div_V = V_r.diff(r) + V_r / r + V_t.diff(t) / r
159
160        if self.mesh.dim == 3:  # Or is this cdim ?
161            z = self.mesh.CoordinateSystem.N[2]
162            V_z = matrix[2]
163            div_V += V_z.diff(z)
164
165        return div_V
166
167    def gradient(self, scalar):
168        r"""
169        $\nabla \phi$
170        """
171
172        if isinstance(scalar, sympy.Matrix) and scalar.shape == (1, 1):
173            scalar = scalar[0, 0]
174
175        grad_S = sympy.Matrix.zeros(1, self.mesh.dim)
176
177        r = self.mesh.CoordinateSystem.N[0]
178        t = self.mesh.CoordinateSystem.N[1]
179
180        grad_S[0] = scalar.diff(r)
181        grad_S[1] = scalar.diff(t) / r
182
183        if self.mesh.dim == 3:  # Or is this cdim ?
184            z = self.mesh.CoordinateSystem.N[2]
185
186            grad_S[2] = scalar.diff(z)
187
188        return grad_S
189
190    def curl(self, matrix):
191        r"""
192        $\nabla \phi$
193        """
194
195        r = self.mesh.CoordinateSystem.N[0]
196        t = self.mesh.CoordinateSystem.N[1]
197
198        matrix0 = self.to_matrix(matrix)
199        V_r = matrix0[0]
200        V_t = matrix0[1]
201
202        # if 2D, return a scalar of the out-of-plane curl
203
204        if self.mesh.dim == 2:
205            curl_V = V_t / r + V_t.diff(r) - V_r.diff(t) / r
206
207        else:
208            z = self.mesh.CoordinateSystem.N[2]
209            V_z = matrix0[2]
210            curl_V = sympy.Matrix.zeros(1, 3)
211
212            curl_V[0] = V_z.diff(t) / r - V_t.diff(z)
213            curl_V[1] = V_r.diff(z) - V_z.diff(r)
214            curl_V[2] = V_t / r + V_t.diff(r) - V_r.diff(t) / r
215
216        return curl_V
217
218    def strain_tensor(self, vector):
219        """
220        Components of the infinitessimal strain or strain-rate tensor where the
221        vector that is provided is displacement or velocity respectively.
222        In cylindrical geometry, there are additional terms that include
223        the location of each point
224        """
225
226        # Coerce vector to sympy.Matrix form
227        matrix = self.to_matrix(vector)
228
229        L = matrix.jacobian(self.mesh.CoordinateSystem.N)
230        r = self.mesh.CoordinateSystem.N[0]
231
232        vr = matrix[0]
233        vt = matrix[1]
234
235        E = L.copy()
236        # E_00, E_22 and E_02 are unchanged from Cartesian
237
238        # E[0,0] = L[0,0]
239        E[1, 1] = L[1, 1] / r + vr / r
240        E[1, 0] = E[0, 1] = (L[0, 1] / r + L[1, 0] - vt / r) / 2
241
242        if self.dim == 3:
243            E[1, 2] = E[2, 1] = (L[2, 1] / r + L[1, 2]) / 2
244            E[0, 2] = E[2, 0] = (L[2, 0] + L[0, 2]) / 2
245
246        return E

387class mesh_vector_calculus_spherical(mesh_vector_calculus):
388    r"""
389    mesh_vector_calculus module for div, grad, curl etc that apply in
390    native spherical coordinates r, \theta, \phi in the standard definition
391    (radius, colatitude, longitude)
392    """
393
394    def __init__(self, mesh):
395        coordinate_type = mesh.CoordinateSystem.coordinate_type
396
397        # validation
398
399        if not coordinate_type == CoordinateSystemType.SPHERICAL_NATIVE:
400            print(
401                f"Warning mesh type {mesh.CoordinateSystem.type} uses Cartesian coordinates not spherical"
402            )
403
404        super().__init__(mesh)
405
406    def divergence(self, matrix):
407        r"""
408        $ \nabla \cdot \mathbf{v} $
409        """
410
411        r = self.mesh.CoordinateSystem.N[0]
412        t = self.mesh.CoordinateSystem.N[1]
413        p = self.mesh.CoordinateSystem.N[2]
414
415        V_r = matrix[0]
416        V_t = matrix[1]
417        V_p = matrix[2]
418
419        # cosecant_th = sympy.Piecewise(
420        #     (1000, sympy.Abs(t) < 0.01 * sympy.pi), (1 / sympy.sin(t), True)
421        # )
422
423        cosec_t = 1 / (sympy.sin(t) + 1.0e-6)
424
425        div_V = (
426            V_r.diff(r)
427            + 2 * V_r / r
428            + cosec_t * (V_p.diff(p) + sympy.cos(t) * V_t) / r
429            + V_t.diff(t) / r
430        )
431
432        return div_V
433
434    def gradient(self, scalar):
435        r"""
436        $\nabla \phi$
437        """
438
439        if isinstance(scalar, sympy.Matrix) and scalar.shape == (1, 1):
440            scalar = scalar[0, 0]
441
442        grad_S = sympy.Matrix.zeros(1, 3)
443
444        r = self.mesh.CoordinateSystem.N[0]
445        t = self.mesh.CoordinateSystem.N[1]
446        p = self.mesh.CoordinateSystem.N[2]
447
448        cosec_t = 1 / (sympy.sin(t) + 1.0e-6)
449
450        grad_S[0] = +scalar.diff(r)
451        grad_S[1] = +scalar.diff(t) / r
452        # grad_S[2] = sympy.Piecewise(
453        #     (1000, sympy.Abs(p) < 0.01 * sympy.pi),
454        #     (scalar.diff(p) / (r * sympy.sin(t)), True),
455        # )
456        grad_S[2] = cosec_t * scalar.diff(p) / r
457
458        return grad_S

460    def curl(self, matrix):
461        r"""
462        $\nabla \times \phi$ in spherical
463        """
464
465        r = self.mesh.CoordinateSystem.N[0]
466        l1 = self.mesh.CoordinateSystem.N[1]
467        l2 = self.mesh.CoordinateSystem.N[2]
468
469        matrix0 = self.to_matrix(matrix)
470        V_r = matrix[0]
471        V_l1 = matrix[1]
472        V_l2 = matrix[2]
473
474        curl_V = sympy.Matrix.zeros(1, 3)
475
476        curl_V[0] = (
477            V_l1.diff(l2) / r
478            - sympy.tan(l2) * V_l1 / r
479            - V_l2.diff(l1) / (r * sympy.cos(l2))
480        )
481        curl_V[1] = V_l2.diff(r) + V_l2 / r - V_r.diff(l2) / r
482        curl_V[2] = V_r.diff(l1) / (r * sympy.cos(l2)) - V_l1.diff(r) - V_l1 / r
483
484        return curl_V

486    def strain_tensor(self, vector):
487        """
488        Components of the infinitessimal strain or strain-rate tensor where the
489        vector that is provided is displacement or velocity respectively.
490        In cylindrical geometry, there are additional terms that include
491        the location of each point
492        """
493
494        # Coerce vector to sympy.Matrix form
495        matrix = self.to_matrix(vector)
496
497        L = matrix.jacobian(self.mesh.CoordinateSystem.N)
498        r = self.mesh.CoordinateSystem.N[0]
499        t = self.mesh.CoordinateSystem.N[1]
500        p = self.mesh.CoordinateSystem.N[2]
501
502        V_r = matrix[0]
503        V_t = matrix[1]
504        V_p = matrix[2]
505
506        cosec_t = 1 / (sympy.sin(t) + 1.0e-6)
507
508        E = L.copy()
509
510        E[0, 0] = L[0, 0]
511        E[1, 1] = (L[1, 1] + V_r) / r
512        E[2, 2] = V_r / r + cosec_t * (L[2, 2] - V_t * sympy.cos(t)) / r
513
514        E[1, 0] = E[0, 1] = (L[0, 1] / r + L[1, 0] - V_t / r) / 2
515        E[1, 2] = E[2, 1] = (L[2, 1] + cosec_t * (L[1, 2] - V_t * sympy.cos(t))) / (
516            2 * r
517        )
518
519        E[2, 0] = E[0, 2] = (cosec_t * L[0, 2] / r + L[2, 0] - V_p / r) / 2
520
521        return E

527    native spherical coordinates on the surface of a sphere (r=r_0).

545    def divergence(self, matrix):
546        r"""
547        $ \nabla \cdot \mathbf{v} $
548        """
549
550        r = sympy.sympify(1)
551        l1 = self.mesh.CoordinateSystem.N[0]
552        l2 = self.mesh.CoordinateSystem.N[1]
553
554        V_r = sympy.sympify(0)
555        V_l1 = matrix[0]
556        V_l2 = matrix[1]
557
558        unstable_term = V_l1.diff(l1) / (r * sympy.cos(l2)) - sympy.tan(l2) * V_l2 / r
559        div_V = (
560            sympy.Piecewise(
561                (0, sympy.Abs(l2) > 0.90 * sympy.pi / 2),
562                (unstable_term, True),
563            )
564            + V_l2.diff(l2) / r
565        )
566
567        return div_V

572        """

614    def strain_tensor(self, vector):
615        """
616        Components of the infinitessimal strain or strain-rate tensor where the
617        vector that is provided is displacement or velocity respectively.
618        In cylindrical geometry, there are additional terms that include
619        the location of each point
620        """
621
622        # Coerce vector to sympy.Matrix form
623        matrix = self.to_matrix(vector)
624
625        L = matrix.jacobian(self.mesh.CoordinateSystem.N)
626        r = sympy.sympify(1)
627        l1 = self.mesh.CoordinateSystem.N[0]
628        l2 = self.mesh.CoordinateSystem.N[1]
629
630        V_r = matrix[0]
631        V_l1 = matrix[1]
632        V_l2 = matrix[2]
633
634        E = L.copy()
635        # E_00, E_22 and E_02 are unchanged from Cartesian
636
637        E[0, 0] = (L[0, 0] - V_l2 * sympy.sin(l2)) / (r * (1.0e-5 + sympy.cos(l2)))
638        E[1, 1] = (L[1, 1]) / r
