165    def update_pre_solve(
166        self,
167        dt: float,
168        evalf: Optional[bool] = False,
169        verbose: Optional[bool] = False,
170        dt_physical: Optional[float] = None,
171    ):
172
173        ## Progress from the oldest part of the history
174        # 1. Copy the stored values down the chain
175
176        if dt_physical is not None:
177            phi = min(1, dt / dt_physical)
178        else:
179            phi = sympy.sympify(1)
180
181        for i in range(self.order - 1, 0, -1):
182            with self.mesh.access(self.psi_star[i]):
183                self.psi_star[i].data[...] = (
184                    phi * self.psi_star[i - 1].data[...]
185                    + (1 - phi) * self.psi_star[i].data[...]
186                )
187
188        # 2. Compute the upstream values
189
190        # We use the u_star variable as a working value here so we have to work backwards
191        for i in range(self.order - 1, -1, -1):
192            with self._nswarm_psi.access(self._nswarm_psi._X0):
193                self._nswarm_psi._X0.data[...] = self._nswarm_psi.data[...]
194
195            # march nodes backwards along characteristics
196            self._nswarm_psi.advection(
197                self.V_fn,
198                -dt,
199                order=2,
200                corrector=False,
201                restore_points_to_domain_func=self.mesh.return_coords_to_bounds,
202                evalf=evalf,
203            )
204
205            if i == 0:
206                # Recalculate u_star from u_fn
207                self._psi_star_projection_solver.uw_function = self.psi_fn
208                self._psi_star_projection_solver.solve()
209
210            if evalf:
211                with self._nswarm_psi.access(self._nswarm_psi.swarmVariable):
212                    for d in range(self.psi_star[i].shape[1]):
213                        self._nswarm_psi.swarmVariable.data[:, d] = uw.function.evalf(
214                            self.psi_star[i].sym[d], self._nswarm_psi.data
215                        )
216            else:
217                with self._nswarm_psi.access(self._nswarm_psi.swarmVariable):
218                    for d in range(self.psi_star[i].shape[1]):
219                        self._nswarm_psi.swarmVariable.data[:, d] = (
220                            uw.function.evaluate(
221                                self.psi_star[i].sym[d], self._nswarm_psi.data
222                            )
223                        )
224
225            # restore coords (will call dm.migrate after context manager releases)
226            with self._nswarm_psi.access(self._nswarm_psi.particle_coordinates):
227                self._nswarm_psi.data[...] = self._nswarm_psi._X0.data[...]
228
229            # Now project to the mesh using bc's to obtain u_star
230            self._psi_star_projection_solver.uw_function = (
231                self._nswarm_psi.swarmVariable.sym
232            )
233
234            self._psi_star_projection_solver.solve()
235
236            # Copy data from the projection operator if required
237            if i != 0:
238                with self.mesh.access(self.psi_star[i]):
239                    self.psi_star[i].data[...] = self.psi_star[0].data[...]
240
241        return

243    def bdf(self, order=None):
244        r"""Backwards differentiation form for calculating DuDt
245        Note that you will need `bdf` / $\delta t$ in computing derivatives"""
246
247        if order is None:
248            order = self.order
249        else:
250            order = max(1, min(self.order, order))
251
252        with sympy.core.evaluate(False):
253            if order == 1:
254                bdf0 = self.psi_fn - self.psi_star[0].sym
255
256            elif order == 2:
257                bdf0 = (
258                    3 * self.psi_fn / 2
259                    - 2 * self.psi_star[0].sym
260                    + self.psi_star[1].sym / 2
261                )
262
263            elif order == 3:
264                bdf0 = (
265                    11 * self.psi_fn / 6
266                    - 3 * self.psi_star[0].sym
267                    + 3 * self.psi_star[1].sym / 2
268                    - self.psi_star[2].sym / 3
269                )
270
271        return bdf0

273    def adams_moulton_flux(self, order=None):
274        if order is None:
275            order = self.order
276        else:
277            order = max(1, min(self.order, order))
278
279        with sympy.core.evaluate(False):
280            if order == 1:
281                am = (self.psi_fn + self.psi_star[0].sym) / 2
282
283            elif order == 2:
284                am = (
285                    5 * self.psi_fn + 8 * self.psi_star[0].sym - self.psi_star[1].sym
286                ) / 12
287
288            elif order == 3:
289                am = (
290                    9 * self.psi_fn
291                    + 19 * self.psi_star[0].sym
292                    - 5 * self.psi_star[1].sym
293                    + self.psi_star[2].sym
294                ) / 24
295
296        return am
