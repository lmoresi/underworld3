365    def write(
366        self,
367        hdfFilename,
368        topologyPath,
369        numCells,
370        numCorners,
371        cellDim,
372        htopologyPath,
373        numHCells,
374        numHCorners,
375        geometryPath,
376        numVertices,
377        spaceDim,
378        time,
379        vfields,
380        cfields,
381        numParticles,
382        pfields,
383    ):
384        self.numVertices = numVertices
385        self.numCells = numCells
386
387        useTime = not (len(time) < 2 and time[0] == -1)
388        with open(self.filename, "w") as fp:
389            self.writeHeader(fp, hdfFilename)
390            # Field information
391            self.writeCells(fp, topologyPath, numCells, numCorners)
392            if numHCells:
393                self.writeCells(fp, htopologyPath, numHCells, numHCorners, "hcells")
394            self.writeVertices(fp, geometryPath, numVertices, spaceDim)
395            if useTime:
396                self.writeTimeGridHeader(fp, time)
397            for t in range(len(time)):
398                if numHCells:
399                    self.writeHybridSpaceGridHeader(fp)
400                    self.writeSpaceGridHeader(
401                        fp, numHCells, numHCorners, cellDim, spaceDim, "hcells"
402                    )
403                    self.writeSpaceGridFooter(fp)
404                self.writeSpaceGridHeader(fp, numCells, numCorners, cellDim, spaceDim)
405                for vf in vfields:
406                    self.writeField(
407                        fp,
408                        len(time),
409                        t,
410                        cellDim,
411                        spaceDim,
412                        "/vertex_fields/" + vf[0],
413                        vf,
414                        "Node",
415                    )
416                for cf in cfields:
417                    self.writeField(
418                        fp,
419                        len(time),
420                        t,
421                        cellDim,
422                        spaceDim,
423                        "/cell_fields/" + cf[0],
424                        cf,
425                        "Cell",
426                    )
427                self.writeSpaceGridFooter(fp)
428                if numHCells:
429                    self.writeSpaceGridFooter(fp)
430            if useTime:
431                self.writeTimeGridFooter(fp)
432            if numParticles:
433                self.writeLocations(fp, numParticles, spaceDim)
434                if useTime:
435                    self.writeTimeGridHeader(fp, time)
436                for t in range(len(time)):
437                    self.writeParticleGridHeader(fp, numParticles, spaceDim)
438                    for pf in pfields:
439                        self.writeParticleField(
440                            fp, pf[0], numParticles, int(pf[1].attrs["Nc"])
441                        )
442                    self.writeSpaceGridFooter(fp)
443                if useTime:
444                    self.writeTimeGridFooter(fp)
445            self.writeFooter(fp)
446        return

449def generateXdmf(hdfFilename, xdmfFilename=None):
450    if xdmfFilename is None:
451        xdmfFilename = os.path.splitext(hdfFilename)[0] + ".xmf"
452    # Read mesh
453    h5 = h5py.File(hdfFilename, "r")
454    if "viz" in h5 and "geometry" in h5["viz"]:
455        geomPath = "viz/geometry"

457    else:
458        geomPath = "geometry"
459        geom = h5["geometry"]
460
461    if "viz" in h5 and "topology" in h5["viz"]:
462        topoPath = "viz/topology"
463        topo = h5["viz"]["topology"]

465        topoPath = "topology"
466        topo = h5["topology"]
467
468    if "viz" in h5 and "hybrid_topology" in h5["viz"]:
469        htopoPath = "viz/hybrid_topology"
470        htopo = h5["viz"]["hybrid_topology"]
471    else:
472        htopoPath = None
473        htopo = None
474
475    vertices = geom["vertices"]
476    numVertices = vertices.shape[0]
477    spaceDim = vertices.shape[1]
478    cells = topo["cells"]
479    numCells = cells.shape[0]
480    numCorners = cells.shape[1]
481    cellDim = topo["cells"].attrs["cell_dim"]
482    if htopo:

485        numHCorners = hcells.shape[1]
486    else:
487        numHCells = 0
488        numHCorners = 0
489    if "time" in h5:
490        time = np.array(h5["time"]).flatten()
491    else:
492        time = [-1]
493    vfields = []
494    cfields = []
495    pfields = []
496    pfields = []
497    if "vertex_fields" in h5:
498        vfields = h5["vertex_fields"].items()
499    if "cell_fields" in h5:
500        cfields = h5["cell_fields"].items()
501    numParticles = 0
502    if "particles" in h5:
503        numParticles = h5["particles"]["coordinates"].shape[0]
504    if "particle_fields" in h5:
505        pfields = h5["particle_fields"].items()
506
507    # Write Xdmf
508    Xdmf(xdmfFilename).write(
509        hdfFilename,
510        topoPath,
511        numCells,
512        numCorners,
513        cellDim,
514        htopoPath,
515        numHCells,
516        numHCorners,

518        numVertices,
519        spaceDim,
520        time,
521        vfields,
522        cfields,
523        numParticles,
524        pfields,

403                    )
404                    self.writeSpaceGridFooter(fp)
405                self.writeSpaceGridHeader(fp, numCells, numCorners, cellDim, spaceDim)
406                for vf in vfields:

409                        len(time),

411                        cellDim,
412                        spaceDim,

421                        t,
422                        cellDim,
423                        spaceDim,
424                        "/cell_fields/" + cf[0],
425                        cf,
426                        "Cell",
427                    )
428                self.writeSpaceGridFooter(fp)
429                if numHCells:
430                    self.writeSpaceGridFooter(fp)
431            if useTime:
432                self.writeTimeGridFooter(fp)
433            if numParticles:
434                self.writeLocations(fp, numParticles, spaceDim)

438                    self.writeParticleGridHeader(fp, numParticles, spaceDim)
439                    for pf in pfields:
